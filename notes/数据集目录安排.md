## 文件目录安排方案

根据你的 `TextDataset` 设计，有多种文件组织方式，我列出从简单到复杂的方案：

---

## 方案1：文本文件 + 标签文件（最简单）

### 目录结构
```
data/
├── texts.txt          # 每行一个文本
└── labels.txt         # 每行一个标签（与 texts.txt 对应）
```

### 完整代码
```python
import torch
from torch.utils.data import Dataset, DataLoader, random_split

class TextDataset(Dataset):
    def __init__(self, texts, labels):
        self.texts = texts
        self.labels = labels
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        return self.texts[idx], self.labels[idx]

def load_data_from_txt(data_dir="data"):
    """从 txt 文件加载数据"""
    with open(f"{data_dir}/texts.txt", "r", encoding="utf-8") as f:
        texts = [line.strip() for line in f.readlines()]
    
    with open(f"{data_dir}/labels.txt", "r", encoding="utf-8") as f:
        labels = [int(line.strip()) for line in f.readlines()]
    
    return texts, labels

# 使用
texts, labels = load_data_from_txt("data")
full_dataset = TextDataset(texts, labels)

# 分割
train_size = int(0.8 * len(full_dataset))
val_size = int(0.1 * len(full_dataset))
test_size = len(full_dataset) - train_size - val_size

train_set, val_set, test_set = random_split(
    full_dataset, 
    [train_size, val_size, test_size],
    generator=torch.Generator().manual_seed(42)
)
```

## 方案2：按类别分文件夹（分类任务常用）

### 目录结构
```
data/
├── neg/                    # 负样本文件夹
│   ├── 0.txt
│   ├── 1.txt
│   ├── 2.txt
│   └── ...
└── pos/                    # 正样本文件夹
    ├── 0.txt
    ├── 1.txt
    ├── 2.txt
    └── ...
```

### 完整代码
```python
import os
import glob

class TextDataset(Dataset):
    def __init__(self, texts, labels):
        self.texts = texts
        self.labels = labels
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        return self.texts[idx], self.labels[idx]

def load_data_from_folders(data_dir="data"):
    """从分类文件夹加载数据"""
    texts = []
    labels = []
    
    # 负样本文件夹 (label=0)
    neg_files = glob.glob(f"{data_dir}/neg/*.txt")
    for file_path in neg_files:
        with open(file_path, "r", encoding="utf-8") as f:
            texts.append(f.read())
            labels.append(0)
    
    # 正样本文件夹 (label=1)
    pos_files = glob.glob(f"{data_dir}/pos/*.txt")
    for file_path in pos_files:
        with open(file_path, "r", encoding="utf-8") as f:
            texts.append(f.read())
            labels.append(1)
    
    return texts, labels

# 使用
texts, labels = load_data_from_folders("data")
full_dataset = TextDataset(texts, labels)
```

## 方案3：CSV/TSV 文件（最通用）

### 目录结构
```
data/
├── train.csv              # 包含 text 和 label 列
├── val.csv                # 可选：预分割的验证集
└── test.csv               # 可选：预分割的测试集
```

**train.csv 格式：**
```csv
text,label
"This movie is great!",1
"Terrible film, waste of time",0
"I really enjoyed this",1
```

### 完整代码
```python
import pandas as pd

class TextDataset(Dataset):
    def __init__(self, texts, labels):
        self.texts = texts
        self.labels = labels
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        return self.texts[idx], self.labels[idx]

def load_data_from_csv(csv_path):
    """从 CSV 文件加载数据"""
    df = pd.read_csv(csv_path)
    texts = df['text'].tolist()
    labels = df['label'].tolist()
    return texts, labels

# 方案A：单个文件 + 随机分割
texts, labels = load_data_from_csv("data/train.csv")
full_dataset = TextDataset(texts, labels)

train_set, val_set, test_set = random_split(
    full_dataset, [20000, 2500, 2500],
    generator=torch.Generator().manual_seed(42)
)

# 方案B：预分割的文件
train_texts, train_labels = load_data_from_csv("data/train.csv")
val_texts, val_labels = load_data_from_csv("data/val.csv")
test_texts, test_labels = load_data_from_csv("data/test.csv")

train_set = TextDataset(train_texts, train_labels)
val_set = TextDataset(val_texts, val_labels)
test_set = TextDataset(test_texts, test_labels)
```

## 方案4：JSON 文件（结构化数据）

### 目录结构
```
data/
└── imdb.json              # 包含所有数据
```

**imdb.json 格式：**
```json
[
    {"text": "This movie is great!", "label": 1},
    {"text": "Terrible film", "label": 0},
    {"text": "I really enjoyed this", "label": 1}
]
```

### 完整代码
```python
import json

class TextDataset(Dataset):
    def __init__(self, texts, labels):
        self.texts = texts
        self.labels = labels
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        return self.texts[idx], self.labels[idx]

def load_data_from_json(json_path):
    """从 JSON 文件加载数据"""
    with open(json_path, "r", encoding="utf-8") as f:
        data = json.load(f)
    
    texts = [item["text"] for item in data]
    labels = [item["label"] for item in data]
    
    return texts, labels

# 使用
texts, labels = load_data_from_json("data/imdb.json")
full_dataset = TextDataset(texts, labels)

train_set, val_set, test_set = random_split(
    full_dataset, [20000, 2500, 2500],
    generator=torch.Generator().manual_seed(42)
)
```

## 方案5：更复杂的项目结构（推荐）

```
imdb_project/
├── data/
│   ├── raw/                          # 原始数据
│   │   ├── aclImdb/                  # IMDB 原始格式
│   │   │   ├── train/
│   │   │   │   ├── pos/
│   │   │   │   └── neg/
│   │   │   └── test/
│   │   └── imdb_data.csv
│   └── processed/                    # 处理后的数据
│       ├── train.pt                  # 保存的 PyTorch 张量
│       ├── val.pt
│       └── test.pt
├── src/
│   ├── dataset.py                    # TextDataset 类定义
│   ├── data_loader.py                # 数据加载和分割
│   └── train.py                      # 训练脚本
└── config.py                         # 配置文件
```

### src/dataset.py
```python
import torch
from torch.utils.data import Dataset

class TextDataset(Dataset):
    def __init__(self, texts, labels):
        self.texts = texts
        self.labels = labels
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        return self.texts[idx], self.labels[idx]
    
    def save(self, path):
        """保存处理后的数据集"""
        torch.save({
            'texts': self.texts,
            'labels': self.labels
        }, path)
    
    @classmethod
    def load(cls, path):
        """加载已处理的数据集"""
        data = torch.load(path)
        return cls(data['texts'], data['labels'])
```

### src/data_loader.py
```python
import os
import glob
import pandas as pd
from src.dataset import TextDataset
from torch.utils.data import random_split
import torch

def load_imdb_from_raw(data_dir="data/raw/aclImdb"):
    """从原始 IMDB 数据集加载（类似文件夹结构）"""
    texts = []
    labels = []
    
    # 加载训练集
    for label_name, label in [("pos", 1), ("neg", 0)]:
        train_path = os.path.join(data_dir, "train", label_name)
        for file_path in glob.glob(f"{train_path}/*.txt"):
            with open(file_path, "r", encoding="utf-8") as f:
                texts.append(f.read())
                labels.append(label)
    
    return texts, labels

def prepare_datasets(texts, labels, split_ratios=[0.8, 0.1, 0.1], seed=42):
    """准备训练/验证/测试集"""
    full_dataset = TextDataset(texts, labels)
    
    # 计算各集大小
    total = len(full_dataset)
    sizes = [int(ratio * total) for ratio in split_ratios[:-1]]
    sizes.append(total - sum(sizes))
    
    # 分割
    datasets = random_split(
        full_dataset, sizes,
        generator=torch.Generator().manual_seed(seed)
    )
    
    return datasets  # (train_set, val_set, test_set)

# 使用示例
if __name__ == "__main__":
    # 加载数据
    texts, labels = load_imdb_from_raw("data/raw/aclImdb")
    print(f"加载了 {len(texts)} 条数据")
    
    # 分割
    train_set, val_set, test_set = prepare_datasets(texts, labels)
    print(f"训练集: {len(train_set)}, 验证集: {len(val_set)}, 测试集: {len(test_set)}")
    
    # 可选：保存处理后的数据集
    # train_set.save("data/processed/train.pt")
```

## 快速对比总结

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| **txt文件** | 最简单 | 标签分离，易错位 | 快速原型 |
| **文件夹分类** | 直观，IMDB原始格式 | 小文件多，IO慢 | 中小数据集 |
| **CSV** | 通用，Excel可打开 | 文本中有逗号需转义 | 通用推荐 |
| **JSON** | 结构化，灵活 | 文件较大 | 复杂标注 |
| **项目结构** | 工程化，可复现 | 初始搭建成本高 | 正式项目 |

**推荐选择**：
- 快速测试：**CSV** 或 **txt文件**
- 实际项目：**方案5的项目结构**
- 已有IMDB数据：直接用 **Hugging Face datasets**（一行代码）
