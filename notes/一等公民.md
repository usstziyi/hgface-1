```python
from datasets import load_dataset
from transformers import AutoTokenizer
from torch.utils.data import DataLoader

# 加载数据和分词器
dataset = load_dataset("imdb")
tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")

# Tokenize
def tokenize_fn(examples):
    return tokenizer(
        examples["text"], 
        truncation=True, 
        max_length=512
    )

tokenized_dataset = dataset.map(tokenize_fn, batched=True)

# 设置格式
tokenized_dataset.set_format(
    type="torch", 
    columns=["input_ids", "attention_mask", "label"]
)

# 直接使用

    tokenized_dataset["train"],
```

### 方案 B：包装成 PyTorch Dataset

```python
from torch.utils.data import Dataset, DataLoader

class IMDbDataset(Dataset):
    def __init__(self, hf_dataset):
        self.dataset = hf_dataset
    
    def __len__(self):
        return len(self.dataset)
    
    def __getitem__(self, idx):
        return (
            self.dataset[idx]["input_ids"],
            self.dataset[idx]["attention_mask"],
            self.dataset[idx]["label"]
        )

# 包装（此时 dataset 已经 tokenize 过了）
train_dataset = IMDbDataset(tokenized_dataset["train"])
train_loader = DataLoader(train_dataset, batch_size=16, shuffle=True)

# 验证
input_ids, attention_mask, labels = next(iter(train_loader))
print(input_ids.shape)  # torch.Size([16, 512])
```

```python
# 完整的时间线
# Step 1: 原始数据
dataset = load_dataset("imdb")
# 列: ['text', 'label']

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")

tokenized = tokenized.remove_columns(["text", "token_type_ids"])
# 列: ['label', 'input_ids', 'attention_mask']

# Step 5: 设置 PyTorch 格式
tokenized.set_format(type="torch", columns=["input_ids", "attention_mask", "label"])

# Step 6: DataLoader
```

```python
from datasets import load_dataset
from transformers import AutoTokenizer

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")

# 加载 + Tokenize + 设置格式，链式调用
dataset = (load_dataset("imdb")
    .map(lambda x: tokenizer(x["text"], truncation=True, max_length=512), batched=True)
    .remove_columns(["text", "token_type_ids"])
)

dataset.set_format(type="torch", columns=["input_ids", "attention_mask", "label"])
```

## 总结

**原始示例省略了 tokenize 步骤**，因为它关注的是"如何把数据格式转为 PyTorch"。完整的流程必须是：

```
原始文本 → Tokenizer → input_ids + attention_mask → set_format/包装 → DataLoader → 模型
```

**缺失 tokenizer 步骤，set\_format 会失败，因为** **`input_ids`** **列还不存在！**
